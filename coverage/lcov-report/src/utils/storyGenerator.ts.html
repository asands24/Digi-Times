const computeSeed = (prompt: string, fileName: string) => {
  const text = `${prompt}|${fileName}`;
  let hash = 0;
  for (let i = 0; i < text.length; i += 1) {
    hash = (hash << 5) - hash + text.charCodeAt(i);
    hash |= 0;
  }
  return Math.abs(hash);
};
